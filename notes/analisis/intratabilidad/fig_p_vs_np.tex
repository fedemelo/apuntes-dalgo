\centering
\begin{tikzpicture}[scale=0.9, font=\small, thick]

  %% =================== VERTICAL AXIS ====================
  \draw[->] (0, -5.2) -- (0, 6.2);
  \node[rotate=90, font=\itshape\scriptsize] at (0.4, 0) {Complejidad temporal};

  %% ================= LEFT PANEL: P ≠ NP =================

  \node[align=center] at (-6, 5.7)
    {$\mathrm{P} \neq \mathrm{NP}$ \\ {\scriptsize (más probable)}};

  % Outer oval: Problemas decidibles
  \draw (-6, 0) ellipse (3.0 and 4.8);
  \node at (-6, 4.2) {Problemas decidibles};

  % NP oval (lower)
  \draw (-6, -1.0) ellipse (2.1 and 2.8);
  % NP label: above P dome top (-2.38) and below NP-Hard bottom (-0.9)
  \node at (-6, -1.6) {NP};

  % NP-Hard oval (upper, overlapping NP)
  \draw (-6, 1.3) ellipse (1.9 and 2.2);
  % NP-Hard label: above NP top (y=1.8), inside NP-Hard
  \node at (-6, 2.6) {NP-Hard};

  % NP-Complete label: inside NP ∩ NP-Hard
  \node at (-6, 0.45) {NP-Complete};

  % P shape: bottom arc follows NP oval exactly, closed with a dome arc.
  % Junction points at 225° and 315° of NP oval: (-7.485, -2.980) and (-4.515, -2.980).
  % This eliminates any gap between P and NP at the bottom.
  \draw (-7.485, -2.980)
    arc[start angle=225, end angle=315, x radius=2.1, y radius=2.8]
    arc[start angle=0, end angle=180, x radius=1.485, y radius=0.6]
    -- cycle;
  \node at (-6, -3.2) {P};

  %% ================ RIGHT PANEL: P = NP ================

  \node[align=center] at (6, 5.7)
    {$\mathrm{P} = \mathrm{NP}$ \\ {\scriptsize (menos probable)}};

  % Outer oval: Problemas decidibles
  \draw (6, 0) ellipse (3.0 and 4.8);
  \node at (6, 4.2) {Problemas decidibles};

  % NP-Hard oval
  \draw (6, 0.3) ellipse (2.2 and 3.6);
  % NP-Hard label: above P=NP=NPC dome top (-1.114)
  \node at (6, 1.5) {NP-Hard};

  % P=NP=NP-Complete shape: bottom arc follows NP-Hard oval exactly.
  % Junction points at 220° and 320° of NP-Hard oval: (4.315, -2.014) and (7.685, -2.014).
  \draw (4.315, -2.014)
    arc[start angle=220, end angle=320, x radius=2.2, y radius=3.6]
    arc[start angle=0, end angle=180, x radius=1.685, y radius=0.9]
    -- cycle;
  \node[align=center] at (6, -2.1)
    {$\mathrm{P} = \mathrm{NP} =$\\NP-Complete};

\end{tikzpicture}
\caption{Clasificación de los problemas computables por clases de complejidad, para los dos escenarios del problema abierto $\mathrm{P}$ vs.\ $\mathrm{NP}$.}
